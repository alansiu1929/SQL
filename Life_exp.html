proc sql;
create table exp_before as
select country, max(life_expectancy) as max_life, avg(life_expectancy) as avg_life, min(life_expectancy) as min_life
from alan.life
where year < 2000
group by country;
create table exp_after as
select country, max(life_expectancy) as max_life, avg(life_expectancy) as avg_life, min(life_expectancy) as min_life
from alan.life
where year >= 2000
group by country;
create table exp_diff as
select a.country,
(a.max_life-b.max_life) as max_life_diff, (a.max_life-b.max_life)/b.max_life *100 as max_life_per_change,
(a.avg_life-b.avg_life) as avg_life_diff, (a.avg_life-b.avg_life)/b.avg_life *100 as avg_life_per_change,
(a.min_life-b.min_life) as min_life_diff, (a.min_life-b.min_life)/b.min_life *100 as min_life_per_change
from exp_before as b
inner join exp_after as a
on a.country=b.country;
run;
proc sql outobs=10;
select country, max_life, avg_life, min_life
from exp_before
where country not like '%income%'
and country not like '%Europe'
and country not like '%America'
and country not like '%regions'
order by avg_life desc;
run;
proc sql;
select country, max_life, avg_life, min_life from exp_after
where country in ('Oceania', 'Asia', 'Europe', 'Africa')
or country like '%America'
order by avg_life desc;
select country, max_life_per_change, avg_life_per_change, min_life_per_change from exp_diff
where country in ('Oceania', 'Asia', 'Europe', 'Africa')
or country like '%America'
order by country;
run;
proc sql outobs=10;
select country, avg_life_per_change from exp_diff
order by avg_life_per_change desc;
select country, max_life_per_change from exp_diff
order by max_life_per_change desc;
select country, min_life_per_change from exp_diff
order by min_life_per_change desc;
select country, max_life_per_change from exp_diff
order by max_life_per_change;
select country, min_life_per_change from exp_diff
order by min_life_per_change;
